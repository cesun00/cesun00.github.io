Erase-remove idiom

In contiguous storage, delete items that satisfy a given predicate.

 1#include <algorithm>
 2#include <vector>
 3#include <iostream>
 4#include <sstream>
 5
 6void pvec(const std::vector<int> &v) {
 7	std::stringstream ss;
 8	ss<< '{';
 9	for(size_t i = 0; i < v.size(); ++i)
10	{
11		if(i != 0)
12			ss << ", ";
13		ss << v[i];
14	}
15	ss<<'}';
16	std::cout<< ss.str() << std::endl;
17}
18
19int main() {
20	std::vector<int> v{1,4,7,2,1,6,5,1,0,9,1,2,3,8};
21	// by simple value:
22	std::vector<int> a(v);
23	pvec(a);
24	a.erase(std::remove(a.begin(),a.end(),1), a.end());
25	pvec(a);
26	// by predicate:
27	std::vector<int> b(v);
28	pvec(b);
29	b.erase(std::remove_if(b.begin(),b.end(),
30				[](const auto &e) { return e<5;}));
31	pvec(b);
32}

Double snprintf

Deprcated since C++20: use std::format.

Use snprintf with nullptr as first argument to verify the validity of format string.

1template<typename ... Args>
2std::string format(const std::string &format, Args ... args) {
3	int size_s = std::snprintf(nullptr, 0, format.c_str(), args ...) + 1; // Extra space for '\0'
4	assert((size_s > 0) && "ill-formed format string.");
5	std::vector<char> buf(static_cast<unsigned long>(size_s));
6	std::snprintf(buf.data(), buf.size(), format.c_str(), args ...);
7	return {buf.cbegin(), buf.cend() - 1}; // We don't want the '\0' inside
8}
