Mind static data member in class template

All instantiations of a class template with the same template arguments share the same memory of data member:

 1#include <cstdio>
 2
 3template<int N>
 4struct Foo {
 5    static int data[N];
 6};
 7
 8template<int N> int <MAXN>::data[N];
 9
10int main() {
11    printf("%p\n", Foo<120>::data);     // 0x5d50b4264040
12    printf("%p\n", Foo<120>::data);     // 0x5d50b4264040
13    printf("%p\n", Foo<240>::data);     // 0x5d50b4264220
14    printf("%p\n", Foo<240>::data);     // 0x5d50b4264220
15}
